        <script>
            function displayTime() {
                let date = new Date();
                let hours = date.toLocaleString('de-DE', { hour: '2-digit', hour12: true, timeZone: 'Asia/Karachi' }).replace(' Uhr', date.toISOString().substr(13, 6));
                document.getElementById("clock").innerHTML = hours;
            }
            setInterval(displayTime, 1000);
